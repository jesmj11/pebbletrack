import { Button } from "@/components/ui/button"
import { Card, CardContent } from "@/components/ui/card"

  id: string
  name: string
  avatar: string
  level: number
  xp: number
  totalXp: number
}

  id: string
  title: string
  subject: string
  dueDate: string
  priority: "low" | "medium" | "high"
  completed: boolean
  xpReward: number
  studentId: string
  description?: string
  emoji: string
  color: string
}

interface StudentDashboardProps {

  onBack: () => void
}

  const [tasks, setTasks] = useState<Task[]>([
    {
      id: "1",
      title: "Math Practice",
      subject: "Mathematics",
      dueDate: "2024-01-15",
      priority: "high",
      completed: false,

      color: "bg-green-400 hover:bg-green-500 border-green-600",
    },
    {

      subject: "Art",
      dueDate: "2024-01-16",

      studentId: student.id,
      emoji: "üé®",
      color: "bg-purple-400 hover:bg-purple-500 border-purple-600",

      color: "bg-yellow-400 hover:bg-yellow-500 border-yellow-600",
    },

    },
    {
      id: "6",
      title: "Play Outside",
      subject: "Physical Education",
      dueDate: "2024-01-16",
      priority: "low",
      completed: false,
      xpReward: 20,
      studentId: student.id,
      emoji: "‚öΩ",

    },
  ])

  const [showCelebration, setShowCelebration] = useState(false)
  const [celebrationMessage, setCelebrationMessage] = useState("")

  const toggleTaskCompletion = (taskId: string) => {
    setTasks(
      tasks.map((task) => {
        if (task.id === taskId && !task.completed) {
          setCelebrationMessage(`üéâ Great Job! +${task.xpReward} XP! üéâ`)
          setShowCelebration(true)
          setTimeout(() => setShowCelebration(false), 3000)
          return { ...task, completed: true }
        }
        return task
      }),
    )
  }

  const completedTasks = tasks.filter((task) => task.completed).length
  const totalTasks = tasks.length
  const todayXP = tasks.filter((t) => t.completed).reduce((sum, t) => sum + t.xpReward, 0)

  return (
    <div className="min-h-screen bg-gradient-to-br from-yellow-300 via-orange-400 to-pink-400 p-4">
      <div className="max-w-4xl mx-auto">

              <div className="flex justify-between text-lg font-bold mb-2 px-4">
                <span>Progress Today</span>
                <span>
                  {completedTasks}/{totalTasks}
                </span>
              </div>
              <Progress value={(completedTasks / totalTasks) * 100} className="h-6 bg-gray-200" />
            </div>

            {/* XP Display */}
            <div className="mt-4 flex justify-center gap-8">
              <div className="bg-white rounded-2xl px-6 py-3 shadow-lg border-2 border-purple-300">

              </div>

                  <Star className="w-6 h-6 text-blue-500" />
                  <span className="text-2xl font-bold text-blue-600">Level {student.level}</span>
                </div>
              </div>
            </div>
          </CardContent>
        </Card>

        {/* Task Buttons Grid */}
        <div className="grid md:grid-cols-2 gap-6">
          {tasks.map((task) => (
            <button
              key={task.id}
              onClick={() => !task.completed && toggleTaskCompletion(task.id)}
              className={`relative p-8 rounded-3xl border-4 transition-all duration-300 transform hover:scale-105 shadow-lg ${
                task.completed
                  ? "bg-green-500 border-green-700 shadow-green-400"
                  : `${task.color} border-4 shadow-xl hover:shadow-2xl`
              } ${!task.completed ? "hover:scale-105 active:scale-95" : ""}`}
              disabled={task.completed}
            >
              {/* Checkbox */}
              <div className="absolute top-4 left-4">
                <div
                  className={`w-8 h-8 border-4 rounded-lg flex items-center justify-center ${
                    task.completed ? "bg-green-500 border-green-600" : "bg-white border-gray-400"
                  }`}
                >
                  {task.completed && <CheckCircle2 className="w-6 h-6 text-white" />}
                </div>
              </div>

              {/* XP Badge */}
              <div className="absolute top-4 right-4">
                <div className="bg-red-600 text-white px-3 py-1 rounded-full text-sm font-bold flex items-center gap-1 shadow-lg">
                  <Trophy className="w-4 h-4" />
                  {task.xpReward}
                </div>
              </div>

              <div className="flex items-center gap-6 mt-4">
                {/* Large Emoji */}
                <div className="text-8xl">{task.emoji}</div>

                {/* Task Text */}

                    className={`text-3xl font-bold mb-2 ${

                  >

                  </p>
                </div>
              </div>

              {/* Completion Overlay */}

                <div className="absolute inset-0 bg-green-400 bg-opacity-20 rounded-3xl flex items-center justify-center">
                  <div className="bg-green-500 text-white px-6 py-3 rounded-2xl text-2xl font-bold shadow-lg">
                    ‚úÖ DONE!
                  </div>
                </div>
              )}
            </button>
          ))}

              üåü You're Doing AMAZING, {student.name}! üåü
            </h3>
            <p className="text-xl text-white" style={{ fontFamily: "Comic Sans MS, cursive" }}>
              {completedTasks === totalTasks
                ? "üéâ All done! You're a learning superstar! üéâ"
                : `Keep going! ${totalTasks - completedTasks} more to go!`}
            </p>
          </CardContent>
        </Card>
      </div>
    </div>
  )
}
